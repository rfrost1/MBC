mbc_tabulate <- function(varname, data, datadict = DD){
  
  first_row <- which(datadict$Variable_Label==varname & !is.na(datadict$Variable_Label) & !is.na(datadict$RDS.Export.Table))
  dict2 <- datadict[first_row:nrow(datadict),]
  last_row <- min(min(which(!is.na(dict2$Variable_Label))[-1])-1, nrow(dict2))
  dictvar <- dict2[1:last_row,]
  vartype <- datadict$VarType[first_row]
  datavar <- pull(data, varname)
  if(vartype %in% c("cts", "dis")){
    
    codes <- dictvar$Code[!is.na(dictvar$Code)]
    
    print(summary(datavar[!datavar%in%codes]))
    
    if(!is.na(datadict$Code[first_row])){
      
      cat_var <-  rep("Non-NA", nrow(data))
      for(code in codes){
        cat_var[datavar==code] <- dictvar$Code_Description[dictvar$Code==code]
      }
      cat_var[is.na(datavar)] <- NA
      
      assign(paste0(varname, "_cat"), cat_var)
      
      freq(get(paste0(varname, "_cat")), cumul = F, order = "freq")
    }
  } else if(vartype == "cat"){
    
    freq(factor(datavar, levels = dictvar$Code, labels = dictvar$Code_Description), cumul = F, order = "freq")
  }
}

mbc_crosstab <- function(varname1, varname2, data, datadict = DD){
  
  data1 <- get_data(varname1, data = data, datadict = datadict)
  data2 <- get_data(varname2, data = data, datadict = datadict)

  vartype1 <- data1["vartype"]
  vartype2 <- data2["vartype"]

  if(vartype1 == "cat" & vartype2 == "cat"){

    cat_cat_tab(data1 = data1, data2 = data2)

  } else if(vartype1 == "cat" & vartype2 %in% c("dis", "cts")){

    cat_num_tab(catdata = data1, numdata = data2)

  } else if(vartype1 %in% c("dis", "cts") & vartype2 == "cat"){

    cat_num_tab(catdata = data2, numdata = data1)

  } else {

    num_num_tab(data1, data2)

  }
}

get_data <- function(varname, data, datadict){
  
  if(!varname %in% datadict$Variable_Label){
    stop(cat(varname, "not in data dictionary!"))
  }

  first_row <- which(datadict$Variable_Label==varname & !is.na(datadict$Variable_Label) & !is.na(datadict$RDS.Export.Table))
  dict2 <- datadict[first_row:nrow(datadict),]
  last_row <- min(min(which(!is.na(dict2$Variable_Label))[-1])-1, nrow(dict2))
  dictvar <- dict2[1:last_row,]
  vartype <- datadict$VarType[first_row]
  datavar <- pull(data, varname)
  codes <- dictvar$Code
  codedesc <- dictvar$Code_Description

  data <- list(varname = varname, data = datavar, vartype = vartype, codes = codes, codedesc = codedesc)

}

cat_cat_tab <- function(data1, data2){
  
  if(length(data1[["codes"]])>length(data2[["codes"]])){
    
      print(ctable(factor(data1[["data"]], levels = data1[["codes"]], labels = data1[["codedesc"]]),
             factor(data2[["data"]], levels = data2[["codes"]], labels = data2[["codedesc"]]),
             prop = "t", dnn = unlist(c(data1[["varname"]], data2[["varname"]]))))
      
    } else {
      
      print(ctable(factor(data2[["data"]], levels = data2[["codes"]], labels = data2[["codedesc"]]),
             factor(data1[["data"]], levels = data1[["codes"]], labels = data1[["codedesc"]]),
             prop = "t", dnn = c(data2[["varname"]], data1[["varname"]])))
      
    }
}

cat_num_tab <- function(catdata, numdata){
  
  for(code in catdata[["codes"]]){
      
    subdata <- numdata[["data"]][catdata[["data"]]==code]
    subdata <- subdata[!is.na(subdata) & !subdata%in%numdata[["codes"]]]
    cat(numdata[["varname"]], "for", catdata[["varname"]], "=", catdata[["codedesc"]][catdata[["codes"]] == code], ":", collapse = "\n")
    print(summary(subdata), collapse = "\n")
    cat("\n")
    
  }
  
  NAdata <- numdata[["data"]][is.na(catdata[["data"]])]
  NAdata <- NAdata[!is.na(NAdata) & !NAdata%in%numdata[["codes"]]]
  cat(numdata[["varname"]], "for", catdata[["varname"]], "= <NA>", ":", collapse = "\n")
  print(summary(NAdata), collapse = "\n")
  cat("\n")
  
  if(!is.na(numdata[["codes"]][1])){
    
    num_cat <- numdata[["data"]]
    num_cat[!numdata[["data"]] %in% numdata[["codes"]] & !is.na(numdata[["data"]])] <- "Non-NA"
    for(code in numdata[["codes"]]){
      num_cat[numdata[["data"]]==code] <- numdata[["codedesc"]][numdata[["codes"]]==code]
    }
    
    print(ctable(factor(catdata[["data"]], levels = catdata[["codes"]], labels = catdata[["codedesc"]]),
                 num_cat, prop = "t", dnn = c(catdata[["varname"]], paste0(numdata[["varname"]], "_cat"))))
  }
}

num_num_tab <- function(data1, data2){
  
  not_valid <- unique(data1[["data"]]%in%data1[["codes"]] + is.na(data1[["data"]]) +
                        data2[["data"]]%in%data2[["codes"]] + is.na(data2[["data"]]))
  
  not_valid <- which(not_valid > 0 | is.na(not_valid))
  
  plot(data1[["data"]][-not_valid], data2[["data"]][-not_valid], 
       xlab = data1[["varname"]], ylab = data2[["varname"]])
  
  if(!is.na(data1[["codes"]][1]) & !is.na(data2[["codes"]][1])){
    
    cat1 <- data1[["data"]]
    cat1[!data1[["data"]] %in% data1[["codes"]] & !is.na(data1[["data"]])] <- "Non-NA"
    for(code in data1[["codes"]]){
      cat1[data1[["data"]]==code] <- data1[["codedesc"]][data1[["codes"]]==code]
    }
    
    cat2 <- data2[["data"]]
    cat2[!data2[["data"]] %in% data2[["codes"]] & !is.na(data2[["data"]])] <- "Non-NA"
    for(code in data2[["codes"]]){
      cat2[data2[["data"]]==code] <- data2[["codedesc"]][data2[["codes"]]==code]
    }
    
    print(ctable(cat1, cat2, prop = "t", dnn = c(data1[["varname"]], data2[["varname"]])))
    
  }
  
}
