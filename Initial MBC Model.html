rf <- rf |>
  mutate(index_bmi_est = case_when(
    indexage > 60 ~ clean_bmiage60,
    indexage > 40 ~ clean_bmiage40,
    indexage > 20 ~ clean_bmiage20,
    TRUE ~ NA
  ), index_bmi_est_cat = case_when(
    index_bmi_est < 18.5 ~ "Underweight",
    between(index_bmi_est, 18.5, 24.9) ~ "Healthy",
    between(index_bmi_est, 25, 29.9) ~ "Overweight",
    between(index_bmi_est, 30, 39.9) ~ "Obsese",
    index_bmi_est > 40 ~ "Severely Obese",
    TRUE ~ "Unknown"
  ), bmichangeper = case_when(
    indexage > 60 ~ clean_bmiage60/clean_bmiage40,
    indexage > 40 ~ clean_bmiage40/clean_bmiage20,
    indexage > 20 ~ NA,
    TRUE ~ NA
  ), bmichangeper = 100*(bmichangeper - 1),
  familybc_spec = case_when(
    familybc == "Yes" & clean_gparentsbreastcancer == "Yes" ~ "First and Second Degree",
    familybc == "Yes" & clean_gparentsbreastcancer == "No" ~ "First Degree Only",
    familybc == "No" & clean_gparentsbreastcancer == "Yes" ~ "Second Degree Only",
    familybc == "No" & clean_gparentsbreastcancer == "No" ~ "No Family History",
    TRUE ~ "Unknown"
  ))
